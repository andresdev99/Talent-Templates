(function() {

    if (!pass_it["cont"]) {
      out["pass_it"] = {

    } else {
      out["pass_it"] = pass_it;
    }

    lastPageNumber = lastPageNumber/50;
    lastPageNumber = lastPageNumber.toFixed(0);
    lastPageNumber = lastPageNumber * 50;
  
    var dom             = window.location.protocol + "//" + window.location.hostname; 
    var pagConstant     = window.location.pathname.split("be/").shift() + "be/";

  function getDescription(url) {
    var xhrrequest = new XMLHttpRequest();
    xhrrequest.open("GET", window.location.origin + window.location.pathname.split("/fs").shift().trim() + "/fs/searchPagination/318c8bb6f553100021d223d9780d30be/0", false); //URL del ajax que trae la informaci√≥n del job
  
    var response = "";
    xhrrequest.onreadystatechange = function () {
      if (xhrrequest.readyState == 4 && xhrrequest.status == 200) {
        //console.log(xhrrequest.responseText);
        response = xhrrequest.responseText;
      }
    };
  
    xhrrequest.send();
    return response;
  }
